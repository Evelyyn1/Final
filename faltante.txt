function drawStar(x, y, radius, spikes, rotation) {
    ctx.save();
    ctx.beginPath();
    ctx.translate(x, y);
    ctx.rotate(rotation * Math.PI / 180);
    ctx.moveTo(0, -radius);
    for (let i = 0; i < spikes; i++) {
        ctx.lineTo(0, -radius * 0.5);
        ctx.rotate(Math.PI / spikes);
        ctx.lineTo(0, -radius);
        ctx.rotate(Math.PI / spikes);
    }
    ctx.closePath();
    ctx.fillStyle = 'white'; // Cambia el color de la estrella.
    ctx.fill();
    ctx.restore();
}
